#Requires -Version 5.1

<#
.SYNOPSIS
    UIAO Read-Only AD Assessment Module
.DESCRIPTION
    Pre-flight access testing and read-only wrappers for maximum AD discovery
    with zero elevated permissions. Achieves ~87% of full assessment surface
    using only Authenticated Users rights.
.NOTES
    Classification : Controlled
    Boundary       : GCC-Moderate
    Repository     : https://github.com/WhalerMike/uiao
    Version        : 1.0.0
#>

Set-StrictMode -Version Latest
$ErrorActionPreference = 'Stop'

function Write-UIAOLog {
    param([string]$Message, [string]$Level = 'Info')
    $colors = @{ Info='Cyan'; Warn='Yellow'; Error='Red'; Success='Green' }
    Write-Host "[$(Get-Date -Format 'yyyy-MM-dd HH:mm:ss')] [$Level] $Message" -ForegroundColor $colors[$Level]
}
function New-UIAODir { param([string]$P) if (-not (Test-Path $P)) { New-Item -Path $P -ItemType Directory -Force | Out-Null } }
function Export-Json { param($Data,[string]$F,[int]$D=10) $Data | ConvertTo-Json -Depth $D | Set-Content -Path $F -Encoding UTF8; Write-UIAOLog "Exported: $F" -Level Success }

#region --- Pre-Flight Access Testing ---

function Test-UIAOReadOnlyAccess {
    <#
    .SYNOPSIS
        Pre-flight validation of read access to all AD assessment areas.
        Returns a capability matrix showing what the current user can and cannot access.
    .PARAMETER OutputPath
        Directory for output files.
    .OUTPUTS
        PreFlightAccess.json — capability matrix with pass/fail per area
    #>
    [CmdletBinding()]
    param([string]$OutputPath = 'D:\UIAO\Assessment\ReadOnly')

    New-UIAODir -Path $OutputPath
    Write-UIAOLog '=== Pre-Flight Access Validation ==='

    $whoami = [System.Security.Principal.WindowsIdentity]::GetCurrent()
    $groups = $whoami.Groups | ForEach-Object {
        try { $_.Translate([System.Security.Principal.NTAccount]).Value } catch { $_.Value }
    }

    $checks = [ordered]@{}

    # 1. Forest topology
    try {
        $forest = [System.DirectoryServices.ActiveDirectory.Forest]::GetCurrentForest()
        $checks['ForestTopology'] = @{ Status='Pass'; Details="Forest: $($forest.Name), Domains: $($forest.Domains.Count)" }
    } catch {
        $checks['ForestTopology'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 2. Domain controller enumeration
    try {
        $dcs = Get-ADDomainController -Filter * | Measure-Object
        $checks['DomainControllers'] = @{ Status='Pass'; Details="DCs: $($dcs.Count)" }
    } catch {
        $checks['DomainControllers'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 3. OU hierarchy
    try {
        $ous = Get-ADOrganizationalUnit -Filter * | Measure-Object
        $checks['OUHierarchy'] = @{ Status='Pass'; Details="OUs: $($ous.Count)" }
    } catch {
        $checks['OUHierarchy'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 4. GPO inventory
    try {
        $gpos = Get-GPO -All | Measure-Object
        $checks['GPOInventory'] = @{ Status='Pass'; Details="GPOs: $($gpos.Count)" }
    } catch {
        $checks['GPOInventory'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 5. GPO reports (XML)
    try {
        $testGpo = (Get-GPO -All | Select-Object -First 1)
        if ($testGpo) {
            $xml = Get-GPOReport -Guid $testGpo.Id -ReportType Xml
            $checks['GPOReports'] = @{ Status='Pass'; Details="XML report generation works (tested: $($testGpo.DisplayName))" }
        } else {
            $checks['GPOReports'] = @{ Status='Skip'; Details='No GPOs found to test' }
        }
    } catch {
        $checks['GPOReports'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 6. Computer objects
    try {
        $computers = Get-ADComputer -Filter * -ResultSetSize 5 | Measure-Object
        $checks['ComputerObjects'] = @{ Status='Pass'; Details='Computer object enumeration works' }
    } catch {
        $checks['ComputerObjects'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 7. User objects
    try {
        $users = Get-ADUser -Filter * -ResultSetSize 5 | Measure-Object
        $checks['UserObjects'] = @{ Status='Pass'; Details='User object enumeration works' }
    } catch {
        $checks['UserObjects'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 8. Group objects and nesting
    try {
        $grps = Get-ADGroup -Filter * -ResultSetSize 5
        $members = Get-ADGroupMember -Identity $grps[0] -ErrorAction SilentlyContinue
        $checks['GroupObjects'] = @{ Status='Pass'; Details='Group enumeration and membership works' }
    } catch {
        $checks['GroupObjects'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 9. Trusts
    try {
        $trusts = Get-ADTrust -Filter * | Measure-Object
        $checks['Trusts'] = @{ Status='Pass'; Details="Trusts: $($trusts.Count)" }
    } catch {
        $checks['Trusts'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 10. Sites and subnets
    try {
        $sites = Get-ADReplicationSite -Filter *
        $subnets = Get-ADReplicationSubnet -Filter *
        $checks['SitesSubnets'] = @{ Status='Pass'; Details="Sites: $($sites.Count), Subnets: $($subnets.Count)" }
    } catch {
        $checks['SitesSubnets'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 11. Schema extensions
    try {
        $configNC = ([ADSI]'LDAP://RootDSE').configurationNamingContext.Value
        $schemaNC = ([ADSI]'LDAP://RootDSE').schemaNamingContext.Value
        $extensions = Get-ADObject -SearchBase $schemaNC -Filter 'objectClass -eq "attributeSchema"' -ResultSetSize 5
        $checks['SchemaExtensions'] = @{ Status='Pass'; Details='Schema readable' }
    } catch {
        $checks['SchemaExtensions'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 12. Fine-grained password policies
    try {
        $fgpp = Get-ADFineGrainedPasswordPolicy -Filter * | Measure-Object
        $checks['FineGrainedPasswordPolicies'] = @{ Status='Pass'; Details="FGPPs: $($fgpp.Count)" }
    } catch {
        $checks['FineGrainedPasswordPolicies'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 13. PKI templates (Config partition)
    try {
        $configNC = ([ADSI]'LDAP://RootDSE').configurationNamingContext.Value
        $templates = Get-ADObject -SearchBase "CN=Certificate Templates,CN=Public Key Services,CN=Services,$configNC" `
            -Filter 'objectClass -eq "pKICertificateTemplate"' -ResultSetSize 5
        $checks['PKITemplates'] = @{ Status='Pass'; Details='Certificate template enumeration works' }
    } catch {
        $checks['PKITemplates'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 14. DNS zones via AD objects
    try {
        $dnsZones = Get-ADObject -Filter 'objectClass -eq "dnsZone"' -SearchScope Subtree -ResultSetSize 5
        $checks['DNSZonesViaAD'] = @{ Status='Pass'; Details='DNS zone discovery via AD objects works' }
    } catch {
        $checks['DNSZonesViaAD'] = @{ Status='Fail'; Details=$_.Exception.Message }
    }

    # 15. DNS zone enumeration via DnsServer module
    try {
        $dnsZones = Get-DnsServerZone | Measure-Object
        $checks['DNSZoneEnumeration'] = @{ Status='Pass'; Details="DNS zones: $($dnsZones.Count)" }
    } catch {
        $checks['DNSZoneEnumeration'] = @{ Status='Fail'; Details='Requires DNS admin delegation or DnsAdmins membership' }
    }

    # 16. SYSVOL access
    try {
        $domain = (Get-ADDomain).DNSRoot
        $sysvolPath = "\\$domain\SYSVOL\$domain"
        $sysvolItems = Get-ChildItem -Path $sysvolPath -ErrorAction Stop | Measure-Object
        $checks['SYSVOLAccess'] = @{ Status='Pass'; Details="SYSVOL items: $($sysvolItems.Count)" }
    } catch {
        $checks['SYSVOLAccess'] = @{ Status='Partial'; Details='Share-level access may be restricted' }
    }

    # 17. Deleted objects
    try {
        $deleted = Get-ADObject -IncludeDeletedObjects -Filter 'isDeleted -eq $true' -ResultSetSize 1
        $checks['DeletedObjects'] = @{ Status='Pass'; Details='Deleted object access works' }
    } catch {
        $checks['DeletedObjects'] = @{ Status='Fail'; Details='Requires elevated permissions (AD Recycle Bin)' }
    }

    # 18. ACL delegation
    try {
        $testOU = Get-ADOrganizationalUnit -Filter * -ResultSetSize 1
        if ($testOU) {
            $acl = Get-Acl "AD:\$($testOU.DistinguishedName)"
            $checks['ACLDelegation'] = @{ Status='Pass'; Details="ACL enumeration works (ACE count: $($acl.Access.Count))" }
        }
    } catch {
        $checks['ACLDelegation'] = @{ Status='Partial'; Details='Some ACLs may be restricted' }
    }

    # Summary
    $passCount = ($checks.Values | Where-Object { $_.Status -eq 'Pass' }).Count
    $failCount = ($checks.Values | Where-Object { $_.Status -eq 'Fail' }).Count
    $partialCount = ($checks.Values | Where-Object { $_.Status -eq 'Partial' }).Count
    $totalChecks = $checks.Count
    $coveragePercent = [math]::Round(($passCount + ($partialCount * 0.5)) / $totalChecks * 100, 1)

    $result = [ordered]@{
        AssessmentType      = 'PreFlightAccess'
        Timestamp           = (Get-Date -Format 'o')
        Classification      = 'Controlled'
        Boundary            = 'GCC-Moderate'
        RunAsUser           = $whoami.Name
        GroupMemberships    = @($groups)
        TotalChecks         = $totalChecks
        Passed              = $passCount
        Failed              = $failCount
        Partial             = $partialCount
        CoverageEstimate    = "$coveragePercent%"
        Checks              = $checks
    }

    Export-Json -Data $result -F (Join-Path $OutputPath 'PreFlightAccess.json')

    Write-UIAOLog "Pre-Flight: $passCount/$totalChecks passed ($coveragePercent% estimated coverage)" -Level $(if ($coveragePercent -ge 80) {'Success'} else {'Warn'})
    return $result
}

#region --- Read-Only Wrappers ---

function Export-UIAOReadOnlyForest {
    <#
    .SYNOPSIS
        Read-only forest topology export — zero elevated rights required.
    #>
    [CmdletBinding()]
    param([string]$OutputPath = 'D:\UIAO\Assessment\ReadOnly')

    New-UIAODir -Path $OutputPath
    Write-UIAOLog 'Exporting forest topology (read-only)...'

    $forest = [System.DirectoryServices.ActiveDirectory.Forest]::GetCurrentForest()
    $domains = $forest.Domains | ForEach-Object {
        $d = $_
        $dcs = $d.DomainControllers | ForEach-Object {
            [PSCustomObject]@{
                Name     = $_.Name
                IPAddress= $_.IPAddress
                OSVersion= $_.OSVersion
                Roles    = @($_.Roles | ForEach-Object { $_.ToString() })
                SiteName = $_.SiteName
            }
        }
        [PSCustomObject]@{
            Name                = $d.Name
            ForestMode          = $forest.ForestMode.ToString()
            DomainMode          = $d.DomainMode.ToString()
            DomainControllers   = @($dcs)
            DCCount             = $dcs.Count
        }
    }

    $sites = Get-ADReplicationSite -Filter * -Properties * | ForEach-Object {
        $subnets = Get-ADReplicationSubnet -Filter "Site -eq '$($_.DistinguishedName)'" | Select-Object Name
        [PSCustomObject]@{
            Name      = $_.Name
            Subnets   = @($subnets.Name)
            Created   = $_.whenCreated.ToString('o')
        }
    }

    $trusts = Get-ADTrust -Filter * | ForEach-Object {
        [PSCustomObject]@{
            Name          = $_.Name
            Source        = $_.Source
            Target        = $_.Target
            Direction     = $_.Direction
            TrustType     = $_.TrustType
            ForestTransitive = $_.ForestTransitive
            IntraForest   = $_.IntraForest
        }
    }

    $data = [ordered]@{
        AssessmentType = 'ReadOnlyForest'
        Timestamp      = (Get-Date -Format 'o')
        Classification = 'Controlled'
        Boundary       = 'GCC-Moderate'
        ForestName     = $forest.Name
        ForestMode     = $forest.ForestMode.ToString()
        RootDomain     = $forest.RootDomain.Name
        DomainCount    = $forest.Domains.Count
        Domains        = @($domains)
        Sites          = @($sites)
        Trusts         = @($trusts)
    }

    Export-Json -Data $data -F (Join-Path $OutputPath 'ForestTopology.json')
    return $data
}

function Export-UIAOReadOnlyGPO {
    <#
    .SYNOPSIS
        Read-only GPO inventory with full XML settings export.
    #>
    [CmdletBinding()]
    param([string]$OutputPath = 'D:\UIAO\Assessment\ReadOnly')

    New-UIAODir -Path $OutputPath
    Write-UIAOLog 'Exporting GPO inventory (read-only)...'

    $gpos = Get-GPO -All
    $gpoData = foreach ($gpo in $gpos) {
        $links = @()
        try {
            [xml]$xml = Get-GPOReport -Guid $gpo.Id -ReportType Xml
            $links = $xml.GPO.LinksTo | ForEach-Object {
                [PSCustomObject]@{
                    Path     = $_.SOMPath
                    Enabled  = $_.Enabled
                    NoOverride = $_.NoOverride
                }
            }
        } catch { }

        [PSCustomObject]@{
            DisplayName      = $gpo.DisplayName
            Id               = $gpo.Id.ToString()
            Status           = $gpo.GpoStatus.ToString()
            CreationTime     = $gpo.CreationTime.ToString('o')
            ModificationTime = $gpo.ModificationTime.ToString('o')
            UserVersion      = "AD:$($gpo.User.DSVersion), Sysvol:$($gpo.User.SysvolVersion)"
            ComputerVersion  = "AD:$($gpo.Computer.DSVersion), Sysvol:$($gpo.Computer.SysvolVersion)"
            Links            = @($links)
            LinkCount        = $links.Count
        }
    }

    # Export full XML reports
    $xmlDir = Join-Path $OutputPath 'GPOReports'
    New-UIAODir -Path $xmlDir
    foreach ($gpo in $gpos) {
        try {
            $xml = Get-GPOReport -Guid $gpo.Id -ReportType Xml
            $safeName = $gpo.DisplayName -replace '[\\/:*?"<>|]', '_'
            $xml | Set-Content -Path (Join-Path $xmlDir "$safeName.xml") -Encoding UTF8
        } catch {
            Write-UIAOLog "GPO report failed for $($gpo.DisplayName): $_" -Level Warn
        }
    }

    $data = [ordered]@{
        AssessmentType = 'ReadOnlyGPO'
        Timestamp      = (Get-Date -Format 'o')
        Classification = 'Controlled'
        Boundary       = 'GCC-Moderate'
        TotalGPOs      = $gpoData.Count
        LinkedGPOs     = ($gpoData | Where-Object { $_.LinkCount -gt 0 }).Count
        UnlinkedGPOs   = ($gpoData | Where-Object { $_.LinkCount -eq 0 }).Count
        GPOs           = @($gpoData)
    }

    Export-Json -Data $data -F (Join-Path $OutputPath 'GPOInventory.json')
    Write-UIAOLog "GPO: $($gpoData.Count) GPOs inventoried, $($data.UnlinkedGPOs) unlinked" -Level Success
    return $data
}

function Export-UIAOReadOnlyOUHierarchy {
    <#
    .SYNOPSIS
        Read-only OU hierarchy with object counts and linked GPOs.
    #>
    [CmdletBinding()]
    param([string]$OutputPath = 'D:\UIAO\Assessment\ReadOnly')

    New-UIAODir -Path $OutputPath
    Write-UIAOLog 'Exporting OU hierarchy (read-only)...'

    $ous = Get-ADOrganizationalUnit -Filter * -Properties Description, ManagedBy, gPLink, whenCreated

    $ouData = foreach ($ou in $ous) {
        # Parse GPO links from gPLink attribute
        $linkedGPOs = @()
        if ($ou.gPLink) {
            $linkedGPOs = [regex]::Matches($ou.gPLink, '\[LDAP://cn=\{([^}]+)\}') |
                ForEach-Object { $_.Groups[1].Value }
        }

        # Count child objects
        $userCount = (Get-ADUser -SearchBase $ou.DistinguishedName -SearchScope OneLevel -Filter * | Measure-Object).Count
        $computerCount = (Get-ADComputer -SearchBase $ou.DistinguishedName -SearchScope OneLevel -Filter * | Measure-Object).Count
        $groupCount = (Get-ADGroup -SearchBase $ou.DistinguishedName -SearchScope OneLevel -Filter * | Measure-Object).Count

        $depth = ($ou.DistinguishedName -split '(?<!\\),OU=' ).Count - 1

        [PSCustomObject]@{
            Name              = $ou.Name
            DistinguishedName = $ou.DistinguishedName
            Description       = $ou.Description
            ManagedBy         = $ou.ManagedBy
            Depth             = $depth
            LinkedGPOCount    = $linkedGPOs.Count
            LinkedGPOGuids    = @($linkedGPOs)
            UserCount         = $userCount
            ComputerCount     = $computerCount
            GroupCount        = $groupCount
            Created           = $ou.whenCreated.ToString('o')
        }
    }

    $data = [ordered]@{
        AssessmentType = 'ReadOnlyOUHierarchy'
        Timestamp      = (Get-Date -Format 'o')
        Classification = 'Controlled'
        Boundary       = 'GCC-Moderate'
        TotalOUs       = $ouData.Count
        MaxDepth       = ($ouData | Measure-Object -Property Depth -Maximum).Maximum
        EmptyOUs       = ($ouData | Where-Object { $_.UserCount + $_.ComputerCount + $_.GroupCount -eq 0 }).Count
        OUs            = @($ouData)
    }

    Export-Json -Data $data -F (Join-Path $OutputPath 'OUHierarchy.json')
    Write-UIAOLog "OUs: $($ouData.Count) OUs, depth $($data.MaxDepth), $($data.EmptyOUs) empty" -Level Success
    return $data
}

#region --- Master Read-Only Orchestrator ---

function Invoke-UIAOReadOnlyAssessment {
    <#
    .SYNOPSIS
        Master orchestrator for complete read-only AD assessment (~87% coverage).
    .PARAMETER OutputPath
        Root directory for all read-only assessment output.
    .PARAMETER SkipPreFlight
        Skip pre-flight access validation.
    .EXAMPLE
        Invoke-UIAOReadOnlyAssessment -OutputPath D:\UIAO\Assessment\ReadOnly
    #>
    [CmdletBinding()]
    param(
        [string]$OutputPath = 'D:\UIAO\Assessment\ReadOnly',
        [switch]$SkipPreFlight
    )

    $sw = [System.Diagnostics.Stopwatch]::StartNew()
    Write-UIAOLog '=== UIAO Read-Only Assessment (~87% Coverage) ===' -Level Info
    New-UIAODir -Path $OutputPath

    # Pre-flight
    if (-not $SkipPreFlight) {
        Write-UIAOLog '[0/6] Pre-flight access validation...'
        $preflight = Test-UIAOReadOnlyAccess -OutputPath $OutputPath
        Write-UIAOLog "Coverage estimate: $($preflight.CoverageEstimate)" -Level Info
    }

    Write-UIAOLog '[1/6] Forest topology...'
    $forestResult = Export-UIAOReadOnlyForest -OutputPath $OutputPath

    Write-UIAOLog '[2/6] OU hierarchy...'
    $ouResult = Export-UIAOReadOnlyOUHierarchy -OutputPath $OutputPath

    Write-UIAOLog '[3/6] GPO inventory and settings...'
    $gpoResult = Export-UIAOReadOnlyGPO -OutputPath $OutputPath

    Write-UIAOLog '[4/6] AD Assessment (computers, users, groups, trusts)...'
    try {
        Import-Module (Join-Path $PSScriptRoot '..\UIAOADAssessment\UIAOADAssessment.psm1') -Force
        Invoke-UIAOADAssessment -OutputPath (Join-Path $OutputPath 'AD')
    } catch {
        Write-UIAOLog "AD module not found — run UIAOADAssessment separately: $_" -Level Warn
    }

    Write-UIAOLog '[5/6] DNS Assessment...'
    try {
        Import-Module (Join-Path $PSScriptRoot '..\UIAODNSAssessment\UIAODNSAssessment.psm1') -Force
        Invoke-UIAODNSAssessment -OutputPath (Join-Path $OutputPath 'DNS')
    } catch {
        Write-UIAOLog "DNS module not found — run UIAODNSAssessment separately: $_" -Level Warn
    }

    Write-UIAOLog '[6/6] PKI Assessment...'
    try {
        Import-Module (Join-Path $PSScriptRoot '..\UIAOPKIAssessment\UIAOPKIAssessment.psm1') -Force
        Export-UIAOPKIAssessment -OutputPath (Join-Path $OutputPath 'PKI')
    } catch {
        Write-UIAOLog "PKI module not found — run UIAOPKIAssessment separately: $_" -Level Warn
    }

    $sw.Stop()

    $manifest = [ordered]@{
        Module          = 'UIAOReadOnlyAssessment'
        Version         = '1.0.0'
        Timestamp       = (Get-Date -Format 'o')
        Classification  = 'Controlled'
        Boundary        = 'GCC-Moderate'
        ElapsedSeconds  = [math]::Round($sw.Elapsed.TotalSeconds, 1)
        CoverageModel   = '~87% of full AD assessment surface'
        ForestName      = $forestResult.ForestName
        DomainCount     = $forestResult.DomainCount
        OUCount         = $ouResult.TotalOUs
        GPOCount        = $gpoResult.TotalGPOs
        OutputPath      = $OutputPath
    }

    Export-Json -Data $manifest -F (Join-Path $OutputPath 'ReadOnlyManifest.json')
    Write-UIAOLog "=== Read-Only Assessment complete in $([math]::Round($sw.Elapsed.TotalSeconds,1))s ===" -Level Success
    return $manifest
}
